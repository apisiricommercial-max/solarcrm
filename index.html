-- ============================================================ -- SETUP: ระบบคลังสินค้า (Inventory) + ใบสั่งซื้อ (Purchase Orders) -- เชื่อมกับ boq_catalog (รายการอุปกรณ์เดิม) และ cashflow (บัญชีบริษัท) -- รันไฟล์นี้ทั้งหมดใน Supabase SQL Editor -- ============================================================ -- 1) ใบสั่งซื้อ (หัวบิล) create table if not exists purchase_orders ( id text primary key, user_id uuid not null references auth.users(id) on delete cascade, supplier_name text, order_date date, expected_date date, received_date date, status text not null default 'pending' check (status in ('pending', 'ordered', 'received', 'cancelled')), note text, "createdAt" timestamptz not null default now() ); alter table purchase_orders enable row level security; create policy "owner can manage own purchase orders" on purchase_orders for all using (auth.uid() = user_id) with check (auth.uid() = user_id); -- เพิ่มคอลัมน์แนบรูปใบเสร็จ/สลิป (เก็บเป็น base64 data URL เหมือนรูปสินค้าในคลัง) alter table purchase_orders add column if not exists receipt_image_url text; -- เพิ่มคอลัมน์ติดตามพัสดุผ่าน 17TRACK alter table purchase_orders add column if not exists tracking_number text; alter table purchase_orders add column if not exists tracking_carrier text; alter table purchase_orders add column if not exists tracking_status text; alter table purchase_orders add column if not exists tracking_status_detail text; alter table purchase_orders add column if not exists tracking_updated_at timestamptz; -- 2) รายการสินค้าในแต่ละใบสั่งซื้อ create table if not exists purchase_order_items ( id text primary key, po_id text not null references purchase_orders(id) on delete cascade, catalog_item_id text, -- อ้างอิงถึง boq_catalog.id ถ้าเลือกจากรายการอุปกรณ์เดิม (ไม่บังคับ) item_name text not null, unit text, quantity numeric not null default 1, unit_price numeric not null default 0, received_quantity numeric not null default 0 ); alter table purchase_order_items enable row level security; create policy "owner can manage own po items" on purchase_order_items for all using ( exists (select 1 from purchase_orders po where po.id = purchase_order_items.po_id and po.user_id = auth.uid()) ) with check ( exists (select 1 from purchase_orders po where po.id = purchase_order_items.po_id and po.user_id = auth.uid()) ); -- 3) สต๊อกสินค้าคงเหลือ (อัปเดตอัตโนมัติเมื่อกด "รับของแล้ว" ในใบสั่งซื้อ) create table if not exists inventory_stock ( id text primary key, user_id uuid not null references auth.users(id) on delete cascade, catalog_item_id text, item_name text not null, unit text, quantity_on_hand numeric not null default 0, min_stock_alert numeric default 0, last_unit_price numeric default 0, -- ราคาต่อหน่วยล่าสุดที่รับเข้า ใช้คำนวณมูลค่าสต๊อก (เผื่อไม่ได้ผูกกับรายการอุปกรณ์) image_url text, -- รูปตัวอย่างสินค้า (เก็บเป็น base64 data URL) updated_at timestamptz not null default now(), unique (user_id, item_name) ); alter table inventory_stock enable row level security; create policy "owner can manage own inventory stock" on inventory_stock for all using (auth.uid() = user_id) with check (auth.uid() = user_id); -- ถ้าเคยรันไฟล์นี้แบบไม่มี 2 คอลัมน์นี้ไปแล้ว ให้รันเพิ่มแยกได้เลย (ปลอดภัย รันซ้ำได้): alter table inventory_stock add column if not exists last_unit_price numeric default 0; alter table inventory_stock add column if not exists image_url text; -- ============================================================ -- ระบบ "จองสต๊อก" จากใบเสนอราคาของลูกค้า -- (ไม่กระทบ quantity_on_hand เดิม จนกว่าจะยืนยันตัดสต๊อกจริง — ปลอดภัยกับระบบเดิม 100%) -- ============================================================ -- เพิ่มคอลัมน์จำนวนที่ถูก "จอง" ไว้แล้ว (แยกจากคงเหลือจริง quantity_on_hand) alter table inventory_stock add column if not exists reserved_quantity numeric not null default 0; -- ตารางบันทึกการจอง ผูกกับลูกค้า + ใบเสนอราคาที่มาจอง create table if not exists stock_reservations ( id text primary key, user_id uuid not null references auth.users(id) on delete cascade, customer_id text not null, quote_id text not null, item_name text not null, quantity numeric not null, status text not null default 'reserved' check (status in ('reserved', 'used', 'cancelled')), "createdAt" timestamptz not null default now() ); alter table stock_reservations enable row level security; create policy "owner can manage own stock reservations" on stock_reservations for all using (auth.uid() = user_id) with check (auth.uid() = user_id); -- ============================================================ -- ประวัติการเคลื่อนไหวสต๊อก (audit log) — บันทึกทุกครั้งที่มีการปรับจำนวน -- พร้อมระบุว่าใช้กับงานของลูกค้ารายไหน จำนวนเท่าไหร่ เพื่อความโปร่งใสแบบมืออาชีพ -- ============================================================ create table if not exists stock_movements ( id text primary key, user_id uuid not null references auth.users(id) on delete cascade, item_name text not null, delta numeric not null, -- บวก = เพิ่มเข้า, ลบ = เบิกออก reason text not null default 'manual', note text, customer_id text, customer_name text, "createdAt" timestamptz not null default now() ); alter table stock_movements enable row level security; create policy "owner can manage own stock movements" on stock_movements for all using (auth.uid() = user_id) with check (auth.uid() = user_id); -- ============================================================ -- เพิ่มเวลาที่แน่นอนของแต่ละขั้นตอน (สำหรับหน้ารายละเอียดใบสั่งซื้อแบบไทม์ไลน์) -- ============================================================ alter table purchase_orders add column if not exists ordered_at timestamptz; alter table purchase_orders add column if not exists received_at timestamptz; -- ============================================================ -- เพิ่มรหัสสินค้า (SKU) และหมวดหมู่ สำหรับหน้าสต๊อกสินค้าโฉมใหม่ -- ============================================================ alter table inventory_stock add column if not exists sku text;
